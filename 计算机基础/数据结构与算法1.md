function min(arr){
    let n = arr.length
    if(n===0){
        return 0
    }else if(n===1){
        return arr[0]
    }else{
        for(let i=0;i<n-1;i++){
            if(arr[i+1]<arr[i]){
                return arr[i+1]
            }
        }
    }
}

function min2(arr){
    let n = arr.length
    if(n===0){
        return 0
    }
    let left = 0
    let right = n-1
    while(left<right){
        let mid = Math.floor((left+right)/2)
        if(arr[mid]>arr[right]){
            left = mid+1
        }else{
            right = mid-1
        }
    }
    return arr[left]
}

//空间O(n) 时间O(n)
function repeat(arr){
    let n = arr.length
    let map = {}
    for(let i=0;i<n;i++){
        if(!map[arr[i]]){
            map[arr[i]]=1
        }else{
            return arr[i]
        }
    }
}
//空间O(1) 时间O(n)
function duplicate(arr){
    let n = arr.length
    for(let i=0;i<n;i++){
        if(arr[i]===i){
            continue
        }else{
            if(arr[arr[i]]!==arr[i]){
                [arr[arr[i]],arr[i]]=[arr[i],arr[arr[i]]]
            }else{
                return arr[i]
            }
        }
    }
}

    for(let i=0;i<n;i++){
        let l = 1

        for(let j=0;j<i;j++){

function construct(arr){
    let n = arr.length
    let res = []
    let l = 1
    res[0] = 1
    for(let i=1;i<n;i++){
        l *= arr[i-1]
        res[i] = l       
    }
    let r = 1
    for(let i=n-2;i>=0;i--){
        r *= arr[i+1]
        res[i] = res[i]*r
    }

function permutation(arr,index,res){
    if(index===arr.length){
        res.push(arr.join(''))
    }
    for(let i = index; i< arr.length; i++){
        [arr[i],arr[index]] = [arr[index],arr[i]]
        permutation(arr,index+1,res)
        [arr[i],arr[index]] = [arr[index],arr[i]]
    }
    return res
}
function res(str){
    if(str.length<=0){return null}
    let res = []
    let arr = str.split('')

    res = [...new Set(res)]

    return res
}

function firstNode(head1, head2) {
    if (head1 === null || head2 === null) {
        return null
    }
    let p1 = head1
    let p2 = head2
    let len1 = 0
    let len2 = 0
    while (p1.next) {
        len1++
        p1 = p1.next
    }
    while (p2.next) {
        len2++
        p2 = p2.next
    }
    p1 = head1
    p2 = head2
    if (len1 > len2) {
        let count = len1 - len2
        while (count) {
            p1 = p1.next
            count--
        }
    } else {
        let count = len2 - len1
        while (count) {
            p2 = p2.next
            count--
        }
    }

        p1 = p1.next
        p2 = p2.next
    }
    return p1
}
