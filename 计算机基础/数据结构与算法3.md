```
//递归函数
function clone(target){
	if(typeof(target)!=='object' || target===null){
		return target
    }
    let res
    if (Array.isArray(target)){
        res = []
    }else{
        res = {}
    }
    for(let i in target){
        let value = target[i]
        if(typeof(value)==='object'){
            res[i] = clone(value)
        }else{
            res[i] = value
        }
    }
    return res
}
```

## 2.顺时针打印矩阵(easy)
```
function print(matrix){
	if(matrix.length===0){return []}
	let l = 0
	let r = matrix[0].length-1
	let t = 0
	let b = matrix.length-1
	let res = []
	while(true){
		for(let i=l;i<=r;i++){
			res.push(matrix[t][i])
		}
		if(++t>b){
			break
		}
		for(let i=t;i<=b;i++){
			res.push(matrix[i][r])
		}
		if(--r<l){
			break
		}
		for(let i=r;i>=l;i--){
			res.push(matrix[b][i])
		}
		if(--b<t){
			break
		}
		for(let i=b;i>=t;i--){
			res.push(matrix[i][l])
		}
		if(++l>r){
			break
		}
	}
	return res
}
```

```
//贪心
function max(arr){
    let n = arr.length
    let cur_sum = arr[0]
    let max_sum = 0
    for(let i=1;i<n;i++){
        cur_sum = Math.max(cur_sum+arr[i],arr[i])
        max_sum = Math.max(cur_sum,max_sum)
    }
    return max_sum
}

//动态规划
function dp(arr){
    let n = arr.length
    let dp = []
    dp[0] = arr[0]
    for(let i=1;i<n;i++){
        if(dp[i-1]>0){
            dp[i] = dp[i-1]+arr[i]
        }else{
            dp[i] = arr[i]
        }
    }
    return Math.max(...dp)
}

//分治
function Status(l,r,m,i){
    this.lsum = l
    this.rsum = r
    this.msum = m
    this.isum = i
}
function pushUp(l,r){
    let isum = l.isum+r.isum
    let lsum = Math.max(l.lsum,l.isum+r.lsum)
    let rsum = Math.max(r.rsum,l.rsum+r.isum)
    let msum = Math.max(Math.max(l.msum,r.msum),l.rsum+r.lsum)
    return new Status(lsum,rsum,msum,isum)
}
function getInfo(a,l,r){
    if(l===r){return new Status(a[l],a[l],a[l],a[l])}
    let m = (l+r)>>1
    let lSub = getInfo(a,l,m)
    let rSub = getInfo(a,m+1,r)
    return pushUp(lSub,rSub)
}
function res(arr){
```

```
function uglyNum(n){
    if(n<=1){return 1}
    let res = []
    res[0] = 1
    let a = 0, b = 0 , c = 0

        res[i] = Math.min(res[a]*2,res[b]*3,res[c]*5)
```

```
function onlyOneTime(s){
    let str = s.split('')
    let n = str.length
    let map = new Map()
    for(let i=0;i<n;i++){
        if(!map.has(str[i])){
            map.set(str[i],1)
        }else{
            map.set(str[i],map.get(str[i])+1)
        }
```

## 6.数组中的逆序对(hard)
```
//时间复杂度O(nlogn) 空间复杂度O(n)
function numbersOfCouple(data){
    let n = data.length
    if(n<=1){return 0}
    let copy = new Array(n)
    for(let i=0;i<n;i++){
        copy[i] = data[i]
    }
    let count = merge(data,copy,0,n-1)
    return count
}

function merge(data,copy,start,end){
    if(start===end){
        return 0
    }
    let length = end-start>>1
    let left = merge(copy,data,start,start+length)
    let right = merge(copy,data,start+length+1,end)
    let i = start+length
    let j = end
    let indexCopy = end
    let count = 0
    while(i>=start&&j>=start+length+1){
        if(data[i]>data[j]){
            copy[indexCopy--] = data[i--]
            count+=j-start-length
        }else{
            copy[indexCopy--] = data[j--]
        }
    }
    while(i>=start){
        copy[indexCopy--] = data[i--]
    }
    while(j>start+length+1){
        copy[indexCopy--] = data[j--]
    }
    return left+right+count
}
```

## 9.正则表达式的匹配(hard)
```
function match(A,B){
    let n = A.length
    let m = B.length
    let f = []
    for(let i=0;i<=n;i++){
        f[i] = []
        for(let j=0;j<=m;j++){
            f[i][j] = null
        }
    }
    for(let i=0;i<=n;i++){
        for(let j=0;j<=m;j++){
            if(j===0){
                f[i][j] = i===0
            }else{
                if(B[j-1]!=='*'){
                    if(i>0&&A[i-1]===B[j-1]||B[j-1]==='.'){
                        f[i][j] = f[i-1][j-1]
                    }
                }else{
                    if(j>=2){
                        f[i][j] = f[i][j-2]
                    }
                    if(i>=1&&j>=2&&A[i-1]===B[j-2]||B[j-2]==='.'){
                        f[i][j] = f[i=1][j]
                    }
                }
            }
        }
    }
    return f[n][m]
}
```

```
function isNumber(str){
    map = {}
    map[0] = {' ':0,'s':1,'d':2,'.':4}
    map[1] = {'d':2,'.':4}
    map[2] = {'d':2,'.':3,'e':5,' ':8}
    map[3] = {'d':3,'e':5,' ':8}
    map[4] = {'d':3}
    map[5] = {'s':6,'d':7}
    map[6] = {'d':7}
    map[7] = {'d':7,' ':8}
    map[8] = {' ':8} 
    let p = 0
    let t
    for(let i=0;i<str.length;i++){
        let c = str[i]
        if(c==='0'&&c<='9'){
            t = 'd'
        }else if(c==='+'||c==='-'){
            t = 's'
        }else if(c==='e'||c==='E'){
            t = 'e'
        }else if(c==='.'||c===' '){
            t = c
        }else{
            t = '?'
        }
        if(!map[p].hasOwnProperty(t)){
            return false
        }
        p = map[p][t]
    }
    return p===2||p===3||p===7||p===8
}
```

```
//最大最小堆实现
function maxNumOfWindow(arr,k){
    if(arr.length===0||k===0){
        return []
    }
    let deque = []
    let res = []
    for(let i=1-k,j=0;j<arr.length;i++,j++){
        if(i>0&&deque[0]===arr[i-1]){
            deque.shift()
        }
        while(deque.length!==0&&deque[deque.length-1]<arr[j]){
            deque.pop()
        }
        deque.push(arr[j])
        if(i>=0){
            res[i] = deque[0]
        }
    }
    return res
}
```

## 13.矩阵中的路径(medium)

```
function dfs(board,word,i,j,k){
    if(i>=board.length||i<0||j>=board[0].length||j<0||board[i][j]!==word[k]){
        return false
    }
    if(k===word.length-1){
        return true
    }
    let tmp = board[i][j]
    board[i][j]='/'
    let res = dfs(board,word,i+1,j,k+1)||dfs(board,word,i-1,j,k+1)||
                dfs(board,word,i,j+1,k+1)||dfs(board,word,i,j-1,k+1)
    board[i][j] = tmp
    return res
}
function exist(board,word){
    for(let i=0;i<board.length;i++){
        for(let j=0;j<board[0].length;j++){
            if(dfs(board,word,i,j,0)){
                return true
            }
        }
    }
    return false
}
```

```
//dfs
function moveCount(m,n,k){
    let visited = new Array(m)
    for(let i=0;i<m;i++){
        visited[i] = new Array(n)
    }
    return dfs(0,0)

    function dfs(i,j){
        if(i<0||i>=m||j<0||j>=n||visited[i][j]||sum(i)+sum(j)>k){
            return 0
        }
        visited[i][j] = true
        return 1+dfs(i+1,j)+dfs(i,j+1)
    }

    function sum(x){
        let sum = 0
        while(x!==0){
            sum += x%10
            x = Math.floor(x/10)
        }
        return sum
    }
}

//bfs
function bfs(m,n,k){
    let res = 0
    let queue = []
    let i = 0
    let j = 0
    queue.push([i,j])
    let visited = new Array(m)
    for(let i=0;i<m;i++){
        visited[i] = new Array(n)
    }
    while(queue.length!==0){
        let x = queue.shift()
        i = x[0]
        j = x[1]
        if(i>=m||j>=n||sum(i)+sum(j)>k||visited[i][j]){
            continue
        }
        visited[i][j] = true
        res++
        queue.push([i+1,j])
        queue.push([i,j+1])
    }
    return res

    function sum(x){
        let sum = 0
        while(x!==0){
            sum += x%10
            x = Math.floor(x/10)
        }
        return sum
    }
}
```
